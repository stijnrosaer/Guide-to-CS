	\section{Gitignore}
    \begin{frame}
		\frametitle{.gitignore}
		% \framesubtitle{}
        \begin{enumerate}
        	\item Maak .gitignore file in root van project
        	\item Best toe te voegen (voor nu)
        	\begin{itemize}
        		\item $.idea$
        		\item $\_\_pycache\_\_$
        		\item $venv$
        		\item $env$
        	\end{itemize}
        	\item Ook regex mogelijk (*.txt)
        \end{enumerate}
	\end{frame}
